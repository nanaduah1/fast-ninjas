import { useHttp } from "./use-http";

A component that does something special and renders text in a div.

### Component usage

```js
function MyComponent() {
  const { get } = useHttp();

  useEffect(() => {
    get("https://something.com/api/stuff")
      .then((res) => res.json())
      .then(console.log);
  });

  return <div>Using Http Get</div>;
}
```
