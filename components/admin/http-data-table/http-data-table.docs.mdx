import { HttpDataTable } from "./http-data-table";

A component that does something special and renders text in a div.

### Component usage

```js
function BasicHttpDataTable() {
  const columns = [
    { field: "userId", title: "User Id" },
    { field: "id", title: "Id" },
    { field: "title", title: "Title" },
  ];
  const src = "https://jsonplaceholder.typicode.com/todos";
  return <HttpDataTable src={src} columns={columns} />;
}
```
